[docs]def plot_setup(pole_longitude=177.5, pole_latitude=37.5,
               scale=30,
               bgcolour=(0.88,0.88,0.88,1),
               grid=True,grid_heavy=2,grid_light=0.5,
               land=True,land_name='GreyT',land_resolution='low',
               cube=True,cube_resolution=0.25,
               width=22,height=16):

    # set the region to plot
    projection=ccrs.RotatedPole(pole_longitude=pole_longitude,
                                pole_latitude=pole_latitude)
    # Make the plot
    fig=matplotlib.pyplot.figure(frameon=False,
                                 facecolor=bgcolour)
    # Set size
    fig.set_size_inches(width, height)

    # Set data extent in rotated degrees
    extent=[scale*-1,scale,scale*-1*width/height,
                          scale*width/height]

    # Define the axes
    ax = matplotlib.pyplot.axes([0., 0., 1., 1.],
                                projection=projection)
    ax.set_axis_off()
    ax.set_extent(extent, crs=projection)
    fig.add_axes(ax)
    # Don't show axes on the plot
    matplotlib.pyplot.axis('off')
    ax.get_xaxis().set_visible(False)
    ax.get_yaxis().set_visible(False)

    # Set the background colour
    ax.background_patch.set_facecolor(bgcolour)

    # Add a lat lon grid
    if grid:
        gl_minor=ax.gridlines(linestyle='-',linewidth=0.2,
                              color=(0,0.50,0,0.3),zorder=0)
        gl_minor.xlocator = matplotlib.ticker.FixedLocator(
                                 numpy.arange(-180,180,grid_light))
        gl_minor.ylocator = matplotlib.ticker.FixedLocator(
                                 numpy.arange(-90,90,grid_light))
        gl_major=ax.gridlines(linestyle='-',linewidth=1,
                              color=(0,0.50,0,0.3),zorder=0)
        gl_major.xlocator = matplotlib.ticker.FixedLocator(
                                 numpy.arange(-180,180,grid_heavy))
        gl_major.ylocator = matplotlib.ticker.FixedLocator(
                                 numpy.arange(-90,90,grid_heavy))

    # Add a land image
    if land:
        ax.background_img(name=land_name, 
                          resolution=land_resolution)

    # Dummy Iris cube to use as a plot grid
    if cube:
        cs_iris=iris.coord_systems.RotatedGeogCS(pole_latitude,
                                                 pole_longitude)
        lat_values=numpy.arange(extent[2]-2,extent[3]+2,cube_resolution)
        lat_coord = iris.coords.DimCoord(lat_values,
                                standard_name='latitude',
                                units='degrees_north',
                                coord_system=cs_iris)
        lon_values=numpy.arange(extent[0]-2,extent[1]+2,cube_resolution)
        lon_coord = iris.coords.DimCoord(lon_values,
                                standard_name='longitude',
                                units='degrees_east',
                                coord_system=cs_iris)
        dummy_data = numpy.zeros((len(lat_values), len(lon_values)))
        plot_cube = iris.cube.Cube(dummy_data,
                           dim_coords_and_dims=[(lat_coord, 0),
                                                (lon_coord, 1)])
